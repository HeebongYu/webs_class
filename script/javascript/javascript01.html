01. 변수 : 데이터 저장소

{
    var x = 100; //변수 x에 숫자 100을 저장
    var y = 200; //변수 y에 숫자 200을 저장
    var z = "Javascript"; //변수 z에 문자 Javascript을 저장

    document.write("＊＊＊ 01. 변수 ＊＊＊");
    document.write(x,);
    document.write(y,);
    document.write(z,);
}

02. 변수 : 데이터 저장 + 데이터 변경

{
    let x = 100;
    let y = 200;
    let z = "Javascript";

    x = 300; //변수 x를 숫자 300으로 변경
    y = 400; //변수 y를 숫자 400으로 변경
    z = "jquery"; //변수 z를 문자 jquery으로 변경

    document.write("＊＊＊ 02. 변수＊＊＊");
    document.write(x,);
    document.write(y,);
    document.write(z,);
}

03. 변수 : 데이터 저장 + 데이터 변경 + 데이터 추가

{
    let x = 100;
    let y = 200;
    let z = "Javascript";

    x += 300; //변수 x에 숫자 300을 추가
    y += 400; //변수 y에 숫자 400을 추가
    z += "jquery" //변수 z에 문자 jquery를 추가

    document.write("＊＊＊ 03. 변수 ＊＊＊");
    document.write(x,);
    document.write(y,);
    document.write(z,);
}

04. 변수 : 변수의 종류 : 지역변수 + 전역변수 + 매개변수

{ //후에 공부 일단 패스
    let x = 100;
    let y = 200;
    let z = "Javascript";

    document.write("＊＊＊ 04. 변수 ＊＊＊");
    document.write(x);
    document.write(y);
    document.write(z);
}

05. 상수 : 데이터 저장 + 데이터 변경(X)

{
    const x = 100;
    const y = 200;
    const z = "Javascript";

    x = 300; //변수 x의 숫자 300으로 변경 (X) 그대로 100
    y = 400; //변수 y의 숫자 400으로 변경 (X) 그대로 200
    z = "jquery"; //변수 z의 문자 jquery으로 변경 (X) 그대로 Javascript

    document.write("＊＊＊ 05. 상수 ＊＊＊");
    document.write(x);
    document.write(y);
    document.write(z);
}

06. 배열 : 여러개의 데이터를 저장 : 표현 방법1 : 기본형

{ //기본형 : 모든 내용 작성합니다.
    const arr = new Array();
    arr[0] = 100;
    arr[1] = 200;
    arr[2] = "Javascript";

    document.write("＊＊＊ 06. 배열 ＊＊＊");
    document.write(arr[0]);
    document.write(arr[1]);
    document.write(arr[2]);
}

07. 배열 : 여러개의 데이터를 저장 : 표현 방법2 : 나열형

{ //나열형 : 저장값을 나열하여 작성합니다.
    const arr = new Array(100, 200, "Javascript");

    document.write("＊＊＊ 07. 배열 ＊＊＊");
    document.write(arr[0]);
    document.write(arr[1]);
    document.write(arr[2]);
}

08. 배열 : 여러개의 데이터를 저장 : 표현 방법3 : 압축형

{ // 압축형 : 약속([])을 통하여 배열을 작성합니다.
    const arr = [];
    arr[0] = 100;
    arr[1] = 200;
    arr[2] = "Javascript";

    document.write("＊＊＊ 08. 배열 ＊＊＊");
    document.write(arr[0]);
    document.write(arr[1]);
    document.write(arr[2]);
}

09. 배열 : 여러개의 데이터를 저장 : 표현 방법4 :통합형

{ //통합형 : 나열형과 압축형을 모두 이용하여 작성합니다.
    const arr = [100, 200, "Javascript"];

    document.write("＊＊＊ 09. 배열 ＊＊＊");
    document.write(arr[0]);
    document.write(arr[1]);
    document.write(arr[2]);
}

10. 객체 : 데이터를 저장(키와 값) : 표현 방법1 : 기본형

{ //기본형 : 모든 내용 작성합니다.
    const obj = new Object();
    obj[0] = 100;
    obj[1] = 200;
    obj[2] = "Javascript";

    document.write("＊＊＊ 10. 객체 ＊＊＊");
    document.write(obj[0]);
    document.write(obj[1]);
    document.write(obj[2]);
}

11. 객체 : 데이터를 저장(키와 값) : 표현 방법2 : 나열형

{ //나열형 : 저장값을 나열하여 작성합니다.
    const obj = new Object();
    obj.a = 100;
    obj.b = 200;
    obj.c = "Javascript";

    document.write("＊＊＊ 11. 객체 ＊＊＊");
    document.write(obj.a);
    document.write(obj.b);
    document.write(obj.c);
}

12. 객체 : 데이터를 저장(키와 값) : 표현 방법3 : 압축형

{ // 압축형 : 약속({})을 통하여 객체을 작성합니다.
    const obj = {};
    obj.a = 100;
    obj.b = 200;
    obj.c = "Javascript";

    document.write("＊＊＊ 12. 객체 ＊＊＊");
    document.write(obj.a);
    document.write(obj.b);
    document.write(obj.c);
}

13. 객체 : 데이터를 저장(키와 값) : 표현 방법4 : 통합형

{ //통합형 : 나열형과 압축형을 모두 이용하여 작성합니다.
    const obj = {a:100, b:200, c:"Javascript"};

    document.write("＊＊＊ 13. 객체 ＊＊＊");
    document.write(obj.a);
    document.write(obj.b);
    document.write(obj.c);
}
